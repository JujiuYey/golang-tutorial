## 1. 接口基础

### 1.1 接口定义

Go的接口是一种类型，定义了一组方法签名：

```go
// 定义接口
type Animal interface {
    Speak() string    // 方法签名
    Move() string
}

// 实现接口
type Dog struct {
    Name string
}

func (d Dog) Speak() string {
    return "汪汪汪"
}

func (d Dog) Move() string {
    return "跑"
}
```

### 1.2 接口实现

只要类型实现了接口的所有方法，就隐式实现了该接口：

```go
type Animal interface {
    Speak() string
}

type Cat struct{}

func (c Cat) Speak() string {  // 实现了Animal接口
    return "喵喵喵"
}

func main() {
    var animal Animal = Cat{}
    fmt.Println(animal.Speak())  // 喵喵喵
}
```

### 1.3 接口变量

```go
type Shape interface {
    Area() float64
    Perimeter() float64
}

type Rectangle struct {
    Width, Height float64
}

func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

func main() {
    // 接口类型变量可以存储任何实现该接口的类型
    var s Shape = Rectangle{Width: 3, Height: 4}
    fmt.Println("面积:", s.Area())       // 12
    fmt.Println("周长:", s.Perimeter())  // 14
}
```

## 2. 接口组合

### 2.1 接口嵌套

```go
type Reader interface {
    Read(p []byte) (n int, err error)
}

type Writer interface {
    Write(p []byte) (n int, err error)
}

// 组合接口
type ReadWriter interface {
    Reader
    Writer
}
```

### 2.2 接口组合的实际应用

```go
package io

type Reader interface {
    Read(p []byte) (n int, err error)
}

type Closer interface {
    Close() error
}

// 组合
type ReadCloser interface {
    Reader
    Closer
}
```

## 3. 空接口与类型断言

### 3.1 空接口

空接口 `interface{}` 不包含任何方法，所有类型都实现了空接口：

```go
var i interface{} = 100
var s interface{} = "hello"
var arr interface{} = []int{1, 2, 3}

// 用于函数参数
func PrintAny(v interface{}) {
    fmt.Println(v)
}

PrintAny(100)
PrintAny("hello")
PrintAny([]int{1, 2, 3})
```

### 3.2 类型断言

```go
var i interface{} = "hello"

// 方式1: 值, ok
s, ok := i.(string)
if ok {
    fmt.Println(s)  // hello
}

// 方式2: 直接断言（不安全）
s := i.(string)  // 如果类型不对会panic

// 安全的方式
if s, ok := i.(string); ok {
    fmt.Println(s)
}
```

### 3.3 type switch

```go
func PrintType(v interface{}) {
    switch v.(type) {
    case int:
        fmt.Println("整数:", v)
    case string:
        fmt.Println("字符串:", v)
    case float64:
        fmt.Println("浮点数:", v)
    case []int:
        fmt.Println("整数切片:", v)
    default:
        fmt.Println("其他类型")
    }
}
```

## 4. 多态实现

### 4.1 接口实现多态

```go
type Animal interface {
    Speak() string
}

type Dog struct{}
func (Dog) Speak() string { return "汪汪汪" }

type Cat struct{}
func (Cat) Speak() string { return "喵喵喵" }

type Cow struct{}
func (Cow) Speak() string { return "哞哞哞" }

// 多态函数
func MakeSpeak(a Animal) {
    fmt.Println(a.Speak())
}

func main() {
    animals := []Animal{Dog{}, Cat{}, Cow{}}
    for _, a := range animals {
        MakeSpeak(a)
    }
    // 输出: 汪汪汪 喵喵喵 哞哞哞
}
```

### 4.2 错误接口

Go标准库定义的error接口：

```go
type error interface {
    Error() string
}

// 自定义错误
type MyError struct {
    msg string
}

func (e MyError) Error() string {
    return e.msg
}

func raise() error {
    return MyError{"发生错误"}
}
```

## 5. 练习作业

### 练习1: 定义形状接口

```go
package main

import (
    "fmt"
    "math"
)

type Shape interface {
    Area() float64
    Perimeter() float64
}

type Circle struct {
    Radius float64
}

func (c Circle) Area() float64 {
    return math.Pi * c.Radius * c.Radius
}

func (c Circle) Perimeter() float64 {
    return 2 * math.Pi * c.Radius
}

type Rectangle struct {
    Width, Height float64
}

func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

func PrintShapeInfo(s Shape) {
    fmt.Printf("面积: %.2f, 周长: %.2f\n", s.Area(), s.Perimeter())
}

func main() {
    shapes := []Shape{
        Circle{Radius: 5},
        Rectangle{Width: 3, Height: 4},
    }
    
    for _, s := range shapes {
        PrintShapeInfo(s)
    }
}
```

### 练习2: 存储系统

```go
package main

import "fmt"

type Storage interface {
    Save(key string, value string) error
    Load(key string) (string, error)
}

type MemoryStorage struct {
    data map[string]string
}

func NewMemoryStorage() *MemoryStorage {
    return &MemoryStorage{data: make(map[string]string)}
}

func (m *MemoryStorage) Save(key, value string) error {
    m.data[key] = value
    return nil
}

func (m *MemoryStorage) Load(key string) (string, error) {
    if v, ok := m.data[key]; ok {
        return v, nil
    }
    return "", fmt.Errorf("key not found: %s", key)
}

func main() {
    var storage Storage = NewMemoryStorage()
    
    storage.Save("name", "张三")
    storage.Save("age", "25")
    
    name, _ := storage.Load("name")
    fmt.Println("name:", name)
}
```

### 挑战题: 日志系统

```go
package main

import "fmt"

type Logger interface {
    Log(msg string)
}

type ConsoleLogger struct{}

func (ConsoleLogger) Log(msg string) {
    fmt.Println("[CONSOLE]", msg)
}

type FileLogger struct {
    filename string
}

func (f FileLogger) Log(msg string) {
    fmt.Printf("[FILE:%s] %s\n", f.filename, msg)
}

type LoggerAdapter struct {
    logger Logger
}

func (a LoggerAdapter) LogLevel(level, msg string) {
    a.logger.Log(fmt.Sprintf("[%s] %s", level, msg))
}

func main() {
    var logger Logger = ConsoleLogger{}
    
    adapter := LoggerAdapter{logger: logger}
    adapter.LogLevel("INFO", "系统启动")
    adapter.LogLevel("ERROR", "发生错误")
    
    // 切换到文件日志
    adapter.logger = FileLogger{filename: "app.log"}
    adapter.LogLevel("INFO", "记录到文件")
}
```

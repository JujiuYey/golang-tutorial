## 1. 数组

### 1.1 数组声明与初始化

```go
// 声明数组
var arr [5]int              // [0, 0, 0, 0, 0]
var arr = [5]int{1, 2, 3, 4, 5}

// 简短声明
arr := [5]int{1, 2, 3, 4, 5}

// 根据初始化自动推断长度
arr := [...]int{1, 2, 3}    // 长度3
```

### 1.2 数组遍历

```go
arr := [5]int{1, 2, 3, 4, 5}

// 普通for循环
for i := 0; i < len(arr); i++ {
    fmt.Println(arr[i])
}

// for-range遍历
for index, value := range arr {
    fmt.Printf("索引: %d, 值: %d\n", index, value)
}
```

### 1.3 多维数组

```go
// 二维数组
var matrix [3][3]int
matrix[0][0] = 1

arr := [2][3]int{
    {1, 2, 3},
    {4, 5, 6},
}

// 遍历
for i := 0; i < 2; i++ {
    for j := 0; j < 3; j++ {
        fmt.Print(arr[i][j], " ")
    }
    fmt.Println()
}
```

### 1.4 数组是值类型

```go
a := [3]int{1, 2, 3}
b := a                    // 复制整个数组
b[0] = 100               // 修改b不影响a
fmt.Println(a[0])        // 1
fmt.Println(b[0])        // 100
```

## 2. 切片

### 2.1 切片声明

```go
// 声明切片
var s []int               // nil切片
s = []int{1, 2, 3}      // 初始化切片

// make创建切片
s := make([]int, 5)      // len=5, cap=5, 值为[0,0,0,0,0]
s := make([]int, 3, 10)  // len=3, cap=10

// 简短声明
s := []int{1, 2, 3, 4, 5}
```

### 2.2 切片操作

```go
s := []int{1, 2, 3, 4, 5}

// len(): 长度
fmt.Println(len(s))  // 5

// cap(): 容量
fmt.Println(cap(s))  // 5

// append(): 添加元素
s = append(s, 6)    // [1,2,3,4,5,6]
s = append(s, 7, 8)  // [1,2,3,4,5,6,7,8]

// 切片追加切片
a := []int{1, 2}
b := []int{3, 4}
a = append(a, b...)  // [1,2,3,4]
```

### 2.3 切片切片

```go
s := []int{1, 2, 3, 4, 5}

// s[low:high] - 左闭右开
fmt.Println(s[1:3])   // [2, 3]
fmt.Println(s[:3])     // [1, 2, 3]
fmt.Println(s[2:])     // [3, 4, 5]
fmt.Println(s[:])      // [1, 2, 3, 4, 5]
```

### 2.4 切片是引用类型

```go
a := []int{1, 2, 3}
b := a                 // b和a指向同一底层数组
b[0] = 100             // 修改b会影响a
fmt.Println(a[0])      // 100
```

### 2.5 切片追加与扩容

```go
s := make([]int, 0, 2)  // len=0, cap=2
s = append(s, 1)        // 容量够，不扩容
s = append(s, 2)        // 容量够，不扩容
s = append(s, 3)        // 容量翻倍，扩容到4
```

### 2.6 删除元素

```go
s := []int{1, 2, 3, 4, 5}

// 删除索引为2的元素
s = append(s[:2], s[3:]...)  // [1, 2, 4, 5]
```

## 3. Map

### 3.1 Map声明与初始化

```go
// 声明Map
var m map[string]int           // nil Map

// make创建
m := make(map[string]int)       // 空Map
m := map[string]int{
    "a": 1,
    "b": 2,
}
```

### 3.2 Map操作

```go
m := make(map[string]int)

// 添加/修改
m["apple"] = 5
m["banana"] = 3

// 读取
count := m["apple"]             // 5

// 判断键是否存在
value, ok := m["cherry"]
if ok {
    fmt.Println(value)
} else {
    fmt.Println("键不存在")
}

// 删除
delete(m, "banana")

// 长度
fmt.Println(len(m))             // 1
```

### 3.3 Map遍历

```go
m := map[string]int{
    "apple": 5,
    "banana": 3,
    "orange": 2,
}

// 遍历键值对
for key, value := range m {
    fmt.Printf("%s: %d\n", key, value)
}

// 只遍历键
for key := range m {
    fmt.Println(key)
}

// 只遍历值
for _, value := range m {
    fmt.Println(value)
}
```

### 3.4 Map是引用类型

```go
m1 := map[string]int{"a": 1}
m2 := m1                      // 指向同一底层结构
m2["a"] = 100
fmt.Println(m1["a"])          // 100
```

## 4. 结构体

### 4.1 结构体定义

```go
// 定义结构体
type Person struct {
    Name    string
    Age     int
    Gender  string
}

// 实例化
var p Person
p.Name = "张三"
p.Age = 25
p.Gender = "男"

// 简短实例化
p := Person{
    Name: "李四",
    Age: 30,
    Gender: "女",
}

// 按顺序初始化（不推荐）
p := Person{"王五", 28, "男"}
```

### 4.2 结构体指针

```go
p := Person{Name: "张三", Age: 25}

// 方式1: 普通访问
p.Age = 26

// 方式2: 指针访问（Go会自动解引用）
pptr := &p
(*pptr).Age = 26        // 繁琐
pptr.Age = 26           // Go简化，等价于上面
```

### 4.3 嵌套结构体

```go
type Address struct {
    City    string
    Street  string
}

type Person struct {
    Name    string
    Address Address  // 嵌套
}

p := Person{
    Name: "张三",
    Address: Address{
        City: "北京",
        Street: "中关村",
    },
}

fmt.Println(p.Address.City)
```

### 4.4 结构体方法

```go
type Person struct {
    Name string
    Age  int
}

// 值接收者方法
func (p Person) SayHello() {
    fmt.Printf("你好，我是%s\n", p.Name)
}

// 指针接收者方法
func (p *Person) Birthday() {
    p.Age++
}

p := Person{Name: "张三", Age: 25}
p.SayHello()    // 你好，我是张三
p.Birthday()   // Age变成26
```

### 4.5 结构体嵌入（类似继承）

```go
type Animal struct {
    Name string
}

func (a Animal) Eat() {
    fmt.Println("吃东西")
}

type Dog struct {
    Animal           // 嵌入
    Breed string
}

dog := Dog{
    Animal: Animal{Name: "旺财"},
    Breed: "金毛",
}

dog.Eat()             // 调用嵌入的方法
fmt.Println(dog.Name) // 访问嵌入字段
```

## 5. 练习作业

### 练习1: 切片操作

```go
package main

import "fmt"

func main() {
    // 创建切片
    nums := []int{1, 2, 3, 4, 5}
    
    // 添加元素
    nums = append(nums, 6, 7, 8)
    
    // 删除索引2的元素
    nums = append(nums[:2], nums[3:]...)
    
    // 遍历打印
    for i, v := range nums {
        fmt.Printf("索引: %d, 值: %d\n", i, v)
    }
}
```

### 练习2: Map统计

```go
package main

import "fmt"

func main() {
    // 统计单词出现次数
    words := []string{"apple", "banana", "apple", "orange", "banana", "apple"}
    
    wordCount := make(map[string]int)
    for _, word := range words {
        wordCount[word]++
    }
    
for word, count := range wordCount {
        fmt.Printf("%s: %d次\n", word, count)
    }
}
```

### 练习3: 学生管理系统

```go
package main

import (
    "fmt"
)

type Student struct {
    Name  string
    Age   int
    Grade float64
}

func main() {
    students := []Student{
        {"张三", 18, 85.5},
        {"李四", 19, 92.0},
        {"王五", 20, 78.5},
    }
    
    // 计算平均分
    var total float64
    for _, s := range students {
        total += s.Grade
    }
    avg := total / float64(len(students))
    
    fmt.Printf("学生总数: %d\n", len(students))
    fmt.Printf("平均分: %.2f\n", avg)
    
    // 查找最高分学生
    maxGrade := students[0]
    for _, s := range students {
        if s.Grade > maxGrade.Grade {
            maxGrade = s
        }
    }
    fmt.Printf("最高分学生: %s (%.2f分)\n", maxGrade.Name, maxGrade.Grade)
}
```

### 挑战题: 联系人簿

```go
package main

import (
    "fmt"
)

type Contact struct {
    Name    string
    Phone   string
    Email   string
}

type AddressBook struct {
    contacts []Contact
}

func (ab *AddressBook) Add(c Contact) {
    ab.contacts = append(ab.contacts, c)
}

func (ab *AddressBook) Delete(name string) {
    for i, c := range ab.contacts {
        if c.Name == name {
            ab.contacts = append(ab.contacts[:i], ab.contacts[i+1:]...)
            return
        }
    }
}

func (ab *AddressBook) Search(name string) *Contact {
    for i := range ab.contacts {
        if ab.contacts[i].Name == name {
            return &ab.contacts[i]
        }
    }
    return nil
}

func (ab *AddressBook) List() {
    for _, c := range ab.contacts {
        fmt.Printf("%s - %s - %s\n", c.Name, c.Phone, c.Email)
    }
}

func main() {
    book := AddressBook{}
    
    book.Add(Contact{"张三", "13800138000", "zhangsan@example.com"})
    book.Add(Contact{"李四", "13900139000", "lisi@example.com"})
    
    fmt.Println("所有联系人:")
    book.List()
    
    fmt.Println("\n查找张三:")
    if c := book.Search("张三"); c != nil {
        fmt.Printf("找到: %s, %s\n", c.Phone, c.Email)
    }
}
```

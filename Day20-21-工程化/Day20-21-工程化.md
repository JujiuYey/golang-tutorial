## 1. Makefile

### 1.1 Makefile基础

```makefile
# 变量
BINARY_NAME=myapp
BUILD_DIR=build
VERSION=1.0.0

# 默认目标
all: build

# 构建
build:
    go build -o $(BUILD_DIR)/$(BINARY_NAME) .

# 清理
clean:
    rm -rf $(BUILD_DIR)

# 运行
run:
    go run main.go

# 测试
test:
    go test -v ./...

# 安装依赖
deps:
    go mod tidy
    go mod download
```

### 1.2 高级Makefile

```makefile
.PHONY: all build clean test run install deps fmt lint

BINARY_NAME=myapp
BUILD_DIR=build
VERSION=1.0.0
BUILD_TIME=$(shell date -u '+%Y-%m-%d_%H:%M:%S')
LDFLAGS=-ldflags "-X main.Version=$(VERSION) -X main.BuildTime=$(BUILD_TIME)"

# 颜色输出
RED=\033[0;31m
GREEN=\033[0;32m
YELLOW=\033[0;33m
NC=\033[0m

all: deps fmt lint test build

deps:
    @echo "$(YELLOW)安装依赖...$(NC)"
    go mod tidy

fmt:
    @echo "$(YELLOW)格式化代码...$(NC)"
    go fmt ./...

lint:
    @echo "$(YELLOW)代码检查...$(NC)"
    go vet ./...

test:
    @echo "$(YELLOW)运行测试...$(NC)"
    go test -v -race -cover ./...

build:
    @echo "$(GREEN)构建项目...$(NC)"
    mkdir -p $(BUILD_DIR)
    go build $(LDFLAGS) -o $(BUILD_DIR)/$(BINARY_NAME) .

clean:
    @echo "$(YELLOW)清理构建文件...$(NC)"
    rm -rf $(BUILD_DIR)

run: build
    ./$(BUILD_DIR)/$(BINARY_NAME)

install: build
    @echo "$(GREEN)安装应用...$(NC)"
    cp $(BUILD_DIR)/$(BINARY_NAME) /usr/local/bin/
```

## 2. 单元测试

### 2.1 编写测试

```go
package main

// 源文件: calculator.go
func Add(a, b int) int {
    return a + b
}

func Multiply(a, b int) int {
    return a * b
}

// 测试文件: calculator_test.go
package main

import "testing"

func TestAdd(t *testing.T) {
    result := Add(2, 3)
    expected := 5
    
    if result != expected {
        t.Errorf("Add(2, 3) = %d, expected %d", result, expected)
    }
}

func TestMultiply(t *testing.T) {
    result := Multiply(4, 5)
    expected := 20
    
    if result != expected {
        t.Errorf("Multiply(4, 5) = %d, expected %d", result, expected)
    }
}
```

### 2.2 运行测试

```bash
# 运行所有测试
go test ./...

# 运行测试并显示详情
go test -v

# 运行匹配模式的测试
go test -v -run TestAdd

# 显示覆盖率
go test -cover

# 生成覆盖率报告
go test -coverprofile=coverage.out
go tool cover -html=coverage.out -o coverage.html
```

### 2.3 表驱动测试

```go
func TestAdd(t *testing.T) {
    tests := []struct {
        name     string
        a, b     int
        expected int
    }{
        {"正数相加", 2, 3, 5},
        {"负数相加", -1, -1, -2},
        {"零", 0, 5, 5},
        {"大正数", 1000000, 2000000, 3000000},
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            result := Add(tt.a, tt.b)
            if result != tt.expected {
                t.Errorf("Add(%d, %d) = %d, expected %d", tt.a, tt.b, result, tt.expected)
            }
        })
    }
}
```

### 2.4 基准测试

```go
// 基准测试函数名以Benchmark开头
func BenchmarkAdd(b *testing.B) {
    for i := 0; i < b.N; i++ {

    }

// 运行基准测试
go test -bench=.

# 指定运行时间
go test -bench=. -benchtime=5s

# 内存统计
go test -bench=. -benchmem
```

## 3. 性能基准测试

### 3.1 使用testing包

```go
package main

import "testing"

func FibRecursive(n int) int {
    if n <= 1 {
        return n
    }
    return FibRecursive(n-1) + FibRecursive(n-2)
}

func FibLoop(n int) int {
    if n <= 1 {
        return n
    }
    a, b := 0, 1
    for i := 2; i <= n; i++ {
        a, b = b, a+b
    }
    return b
}

func BenchmarkFibRecursive(b *testing.B) {
    for i := 0; i < b.N; i++ {
        FibRecursive(20)
    }
}

func BenchmarkFibLoop(b *testing.B) {
    for i := 0; i < b.N; i++ {
        FibLoop(20)
    }
}
```

## 5. CI/CD

### 5.1 GitHub Actions

```yaml
# .github/workflows/ci.yml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v4
      
      - name: Set up Go
        uses: actions/setup-go@v5
        with:
          go-version: '1.22'
          
      - name: Cache modules
        uses: actions/cache@v3
        with:
          path: ~/go/pkg/mod
          key: ${{ runner.os }}-go-${{ hashFiles('**/go.sum') }}
          
      - name: Install dependencies
        run: go mod download
        
      - name: Format check
        run: |
          go fmt ./...
          git diff --exit-code
          
      - name: Lint
        run: go vet ./...
        
      - name: Test
        run: go test -v -race -coverprofile=coverage.out ./...
        
      - name: Upload coverage
        uses: actions/upload-artifact@v3
        with:
          name: coverage
          path: coverage.out
          
      - name: Build
        run: go build -o myapp .
```

### 5.2 Docker构建

```dockerfile
# Dockerfile
FROM golang:1.22-alpine AS builder

WORKDIR /app

COPY go.mod go.sum ./
RUN go mod download

COPY . .
RUN CGO_ENABLED=0 GOOS=linux go build -o myapp cmd/app/main.go

FROM alpine:latest
RUN apk --no-cache add ca-certificates

WORKDIR /root/
COPY --from=builder /app/myapp .

EXPOSE 8080

CMD ["./myapp"]
```

## 6. 练习作业

### 练习1: 编写测试

```go
// stringutil/reverse.go
package stringutil

func Reverse(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

// stringutil/reverse_test.go
package stringutil

import "testing"

func TestReverse(t *testing.T) {
    tests := []struct {
        input    string
        expected string
    }{
        {"hello", "olleh"},
        {"world", "dlrow"},
        {"", ""},
        {"a", "a"},
        {"中文", "文中"},
    }
    
    for _, tt := range tests {
        result := Reverse(tt.input)
        if result != tt.expected {
            t.Errorf("Reverse(%q) = %q, want %q", tt.input, result, tt.expected)
        }
    }
}
```

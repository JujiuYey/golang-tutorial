## 1. IO操作

### 1.1 文件读写

```go
import (
    "fmt"
    "os"
)

// 读取文件
data, err := os.ReadFile("test.txt")
if err != nil {
    fmt.Println("错误:", err)
    return
}
fmt.Println(string(data))

// 写入文件
content := "Hello, Go!"
err := os.WriteFile("output.txt", []byte(content), 0644)
```

### 1.2 分行读取

```go
import (
    "bufio"
    "fmt"
    "os"
)

file, err := os.Open("test.txt")
if err != nil {
    return
}
defer file.Close()

scanner := bufio.NewScanner(file)
for scanner.Scan() {
    fmt.Println(scanner.Text())
}
```

### 1.3 带缓冲的读写

```go
import (
    "bufio"
    "fmt"
    "os"
)

// 读取
file, _ := os.Open("test.txt")
reader := bufio.NewReader(file)

for {
    line, err := reader.ReadString('\n')
    if err != nil {
        break
    }
    fmt.Print(line)
}

// 写入
file, _ := os.Create("output.txt")
writer := bufio.NewWriter(file)
writer.WriteString("Hello\n")
writer.WriteString("World\n")
writer.Flush()  // 必须flush
```

### 1.4 字符串操作

```go
import (
    "fmt"
    "strings"
)

s := "Hello, World!"

// 包含
fmt.Println(strings.Contains(s, "World"))  // true

// 前缀/后缀
fmt.Println(strings.HasPrefix(s, "Hello"))  // true
fmt.Println(strings.HasSuffix(s, "!"))    // true

// 分割
parts := strings.Split(s, ", ")
fmt.Println(parts)  // [Hello World!]

// 替换
s2 := strings.ReplaceAll(s, "World", "Go")
fmt.Println(s2)  // Hello, Go!

// 大小写
fmt.Println(strings.ToUpper("hello"))  // HELLO
fmt.Println(strings.ToLower("HELLO"))  // hello

// 去除空格
fmt.Println(strings.TrimSpace("  hello  "))  // hello

// 拼接
arr := []string{"a", "b", "c"}
fmt.Println(strings.Join(arr, "-"))  // a-b-c
```

### 1.5 字符串转换

```go
import (
    "fmt"
    "strconv"
)

// int转string
s := strconv.Itoa(100)
s := strconv.FormatInt(100, 10)

// string转int
i, _ := strconv.Atoi("100")
i, _ := strconv.ParseInt("100", 10, 64)

// float转string
s := strconv.FormatFloat(3.14159, 'f', 2, 64)  // 3.14

// string转float
f, _ := strconv.ParseFloat("3.14", 64)

// bool转string
s := strconv.FormatBool(true)
b, _ := strconv.ParseBool("true")
```

## 2. JSON处理

### 2.1 JSON编码

```go
import (
    "encoding/json"
    "fmt"
)

type Person struct {
    Name    string   `json:"name"`
    Age     int      `json:"age"`
    Email   string   `json:"email,omitempty"`
    hobbies []string `json:"hobbies"`
}

func main() {
    p := Person{
        Name:    "张三",
        Age:     25,
        Hobbies: []string{"读书", "编程"},
    }
    
    // 编码为JSON
    data, err := json.Marshal(p)
    if err != nil {
        return
    }
    fmt.Println(string(data))
    // {"name":"张三","age":25,"hobbies":["读书","编程"]}
}
```

### 2.2 JSON解码

```go
jsonStr := `{"name":"张三","age":25,"hobbies":["读书"]}`
var p Person
err := json.Unmarshal([]byte(jsonStr), &p)
if err != nil {
    return
}
fmt.Printf("%+v\n", p)
```

### 2.3 结构体标签

```go
type Person struct {
    Name    string `json:"name"`          // 映射为"name"
    Age     int    `json:"age"`           // 映射为"age"
    Secret  string `json:"-"`             // 忽略此字段
    Title   string `json:"title,omitempty"` // 为空时忽略
}
```

### 2.4 Map与JSON

```go
// Map转JSON
m := map[string]interface{}{
    "name": "张三",
    "age": 25,
}
data,_ := json.Marshal(m)

// JSON转Map
jsonStr := `{"name":"张三","age":25}`
var m2 map[string]interface{}
json.Unmarshal([]byte(jsonStr), &m2)
```

## 3. 时间处理

### 3.1 获取当前时间

```go
import (
    "fmt"
    "time"
)

now := time.Now()
fmt.Println(now)
fmt.Println(now.Format("2006-01-02 15:04:05"))
```

### 3.2 时间格式化

| 格式 | 示例 |
|------|------|
| 2006 | 2006 (年) |
| 01 | 01 (月) |
| 02 | 02 (日) |
| 15 | 15 (小时24h) |
| 04 | 04 (分钟) |
| 05 | 05 (秒) |

```go
now := time.Now()

fmt.Println(now.Format("2006-01-02"))        // 2024-01-15
fmt.Println(now.Format("15:04:05"))          // 14:30:00
fmt.Println(now.Format("2006-01-02 15:04:05")) // 2024-01-15 14:30:00
```

### 3.3 时间运算

```go
now := time.Now()

// 加上1小时
later := now.Add(time.Hour)
later = now.Add(30 * time.Minute)

// 减去1天
yesterday := now.Add(-24 * time.Hour)

// 时间差
diff := later.Sub(now)
fmt.Println(diff.Seconds())  // 3600
```

### 3.4 时间戳转换

```go
now := time.Now()

// 时间转时间戳
timestamp := now.Unix()      // 秒
timestampNano := now.UnixNano()  // 纳秒

// 时间戳转时间
t := time.Unix(timestamp, 0)
fmt.Println(t.Format("2006-01-02 15:04:05"))
```

### 3.5 定时器

```go
// 一次性定时器
timer := time.NewTimer(2 * time.Second)
<-timer.C
fmt.Println("2秒后执行")

// ticker循环定时
ticker := time.NewTicker(1 * time.Second)
for i := 0; i < 5; i++ {
    <-ticker.C
    fmt.Println("每秒执行")
}
ticker.Stop()
```

## 4. Flag命令行参数

### 4.1 基本用法

```go
package main

import (
    "flag"
    "fmt"
)

func main() {
    // 定义命令行参数
    name := flag.String("name", "World", "用户名")
    age := flag.Int("age", 0, "年龄")
    verbose := flag.Bool("v", false, "详细输出")
    
    // 解析参数
    flag.Parse()
    
    fmt.Printf("Hello, %s!\n", *name)
    fmt.Printf("Age: %d\n", *age)
    if *verbose {
        fmt.Println("Verbose mode enabled")
    }
}
```

### 4.2 运行

```bash
go run main.go -name=张三 -age=25 -v
```

### 4.3 其他Flag函数

```go
// Int类型的另一种方式
var port int
flag.IntVar(&port, "port", 8080, "端口号")

// 获取非flag参数
args := flag.Args()  // []string

// 显示帮助
flag.Usage = func() {
    fmt.Println("Usage: ./app [options]")
    flag.PrintDefaults()
}
```

## 5. 练习作业

### 练习1: 文件复制工具

```go
package main

import (
    "fmt"
    "io"
    "os"
)

func main() {
    if len(os.Args) != 3 {
        fmt.Println("Usage: cp source dest")
        return
    }
    
    src, dst := os.Args[1], os.Args[2]
    
    sourceFile, err := os.Open(src)
    if err != nil {
        fmt.Println("无法打开源文件:", err)
        return
    }
    defer sourceFile.Close()
    
    destFile, err := os.Create(dst)
    if err != nil {
        fmt.Println("无法创建目标文件:", err)
        return
    }
    defer destFile.Close()
    
    _, err = io.Copy(destFile, sourceFile)
    if err != nil {
        fmt.Println("复制失败:", err)
        return
    }
    
    fmt.Println("复制成功!")
}
```

### 练习2: JSON配置文件

```go
package main

import (
    "encoding/json"
    "fmt"
    "os"
)

type Config struct {
    AppName string `json:"app_name"`
    Port    int    `json:"port"`
    Debug   bool   `json:"debug"`
}

func main() {
    config := Config{
        AppName: "MyApp",
        Port:    8080,
        Debug:   true,
    }
    
    // 编码
    data, err := json.MarshalIndent(config, "", "  ")
    if err != nil {
        return
    }
    fmt.Println(string(data))
    
    // 写入文件
    os.WriteFile("config.json", data, 0644)
    
    // 读取
    content, _ := os.ReadFile("config.json")
    var cfg Config
    json.Unmarshal(content, &cfg)
    fmt.Printf("%+v\n", cfg)
}
```

### 挑战题: 日志工具

```go
package main

import (
    "flag"
    "fmt"
    "os"
    "time"
)

func main() {
    logFile := flag.String("log", "", "日志文件路径")
    level := flag.String("level", "INFO", "日志级别")
    message := flag.String("msg", "", "日志内容")
    
    flag.Parse()
    
    if *message == "" {
        flag.Usage()
        return
    }
    
    timestamp := time.Now().Format("2006-01-02 15:04:05")
    logLine := fmt.Sprintf("[%s] [%s] %s\n", timestamp, *level, *message)
    
    if *logFile != "" {
        file, err := os.OpenFile(*logFile, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0644)
        if err != nil {
            fmt.Println("打开日志文件失败:", err)
            return
        }
        defer file.Close()
        file.WriteString(logLine)
    } else {
        fmt.Print(logLine)
    }
}
```

## 1. HTTP服务器

### 1.1 简单HTTP服务器

```go
package main

import (
    "fmt"
    "net/http"
)

func main() {
    // 处理函数
    http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintf(w, "Hello, World!")
    })
    
    http.HandleFunc("/about", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintf(w, "About Page")
    })
    
    // 启动服务器
    fmt.Println("服务器启动在 :8080")
    err := http.ListenAndServe(":8080", nil)
    if err != nil {
        fmt.Println("服务器错误:", err)
    }
}
```

### 1.2 使用Handler

```go
type HelloHandler struct{}

func (h HelloHandler) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hello from Handler!")
}

func main() {
    handler := HelloHandler{}
    http.Handle("/hello", handler)
    http.ListenAndServe(":8080", nil)
}
```

### 1.3 ServeMux多路复用器

```go
func home(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Home Page")
}

func about(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "About Page")
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("/", home)
    mux.HandleFunc("/about", about)
    
    http.ListenAndServe(":8080", mux)
}
```

## 2. HTTP客户端

### 2.1 GET请求

```go
package main

import (
    "fmt"
    "io"
    "net/http"
)

func main() {
    resp, err := http.Get("https://api.github.com")
    if err != nil {
        fmt.Println("请求失败:", err)
        return
    }
    defer resp.Body.Close()
    
    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return
    }
    
    fmt.Println("状态码:", resp.StatusCode)
    fmt.Println("响应内容:", string(body))
}
```

### 2.2 POST请求

```go
import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    data := map[string]string{"name": "张三", "age": "25"}
    jsonData, _ := json.Marshal(data)
    
    resp, err := http.Post(
        "https://api.example.com/users",
        "application/json",
        bytes.NewBuffer(jsonData),
    )
    if err != nil {
        return
    }
    defer resp.Body.Close()
    
    fmt.Println("状态码:", resp.StatusCode)
}
```

### 2.3 带Header的请求

```go
import (
    "fmt"
    "net/http"
)

func main() {
    req, _ := http.NewRequest("GET", "https://api.github.com/user", nil)
    req.Header.Set("Authorization", "Bearer YOUR_TOKEN")
    req.Header.Set("Accept", "application/json")
    
    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        return
    }
    defer resp.Body.Close()
    
    fmt.Println("状态码:", resp.StatusCode)
}
```

### 3.3 中间件模式

```go
// 中间件函数
func Logger(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        fmt.Printf("[%s] %s\n", r.Method, r.URL.Path)
        next.ServeHTTP(w, r)
    })
}

func Auth(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        token := r.Header.Get("Authorization")
        if token == "" {
            http.Error(w, "Unauthorized", 401)
            return
        }
        next.ServeHTTP(w, r)
    })
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        w.Write([]byte("Home"))
    })
    
    // 应用中间件
    handler := Logger(Auth(mux))
    http.ListenAndServe(":8080", handler)
}
```

## 4. TCP/UDP编程

### 4.1 TCP服务器

```go
package main

import (
    "fmt"
    "net"
)

func main() {
    listener, err := net.Listen("tcp", ":8080")
    if err != nil {
        fmt.Println("监听失败:", err)
        return
    }
    defer listener.Close()
    
    fmt.Println("TCP服务器启动在 :8080")
    
    for {
        conn, err := listener.Accept()
        if err != nil {
            continue
        }
        
        go handleConnection(conn)
    }
}

func handleConnection(conn net.Conn) {
    defer conn.Close()
    
    buffer := make([]byte, 1024)
    for {
        n, err := conn.Read(buffer)
        if err != nil {
            fmt.Println("读取错误:", err)
            return
        }
        
        fmt.Printf("收到: %s\n", string(buffer[:n]))
        
        // 发送响应
        conn.Write([]byte("Message received!\n"))
    }
}
```

### 4.2 TCP客户端

```go
package main

import (
    "fmt"
    "net"
    "os"
)

func main() {
    conn, err := net.Dial("tcp", "localhost:8080")
    if err != nil {
        fmt.Println("连接失败:", err)
        return
    }
    defer conn.Close()
    
    // 发送数据
    conn.Write([]byte("Hello Server!"))
    
    // 读取响应
    buffer := make([]byte, 1024)
    n, _ := conn.Read(buffer)
    fmt.Printf("收到: %s\n", string(buffer[:n]))
}
```

### 4.3 UDP编程

```go
package main

import (
    "fmt"
    "net"
)

func main() {
    // 服务器
    addr, _ := net.ResolveUDPAddr("udp", ":8080")
    conn, _ := net.ListenUDP("udp", addr)
    defer conn.Close()
    
    buffer := make([]byte, 1024)
    for {
        n, remote, _ := conn.ReadFromUDP(buffer)
        fmt.Printf("收到: %s\n", string(buffer[:n]))
        
        conn.WriteToUDP([]byte("OK"), remote)
    }
}
```

## 5. REST API设计

### 5.1 RESTful原则

| 方法 | 路径 | 说明 |
|------|------|------|
| GET | /users | 获取用户列表 |
| GET | /users/:id | 获取单个用户 |
| POST | /users | 创建用户 |
| PUT | /users/:id | 更新用户 |
| DELETE | /users/:id | 删除用户 |

### 5.2 REST API实现

```go
package main

import (
    "encoding/json"
    "fmt"
    "net/http"
    "sync"
)

type User struct {
    ID   int    `json:"id"`
    Name string `json:"name"`
    Age  int    `json:"age"`
}

var (
    users = map[int]User{1: {1, "张三", 25}}
    nextID = 2
    mu sync.Mutex
)

func getUsers(w http.ResponseWriter, r *http.Request) {
    mu.Lock()
    defer mu.Unlock()
    
    result := make([]User, 0, len(users))
    for _, u := range users {
        result = append(result, u)
    }
    
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(result)
}

func getUser(w http.ResponseWriter, r *http.Request) {
    mu.Lock()
    defer mu.Unlock()
    
    id := 1 // 从URL获取
    if user, ok := users[id]; ok {
        json.NewEncoder(w).Encode(user)
    } else {
        http.Error(w, "User not found", 404)
    }
}

func createUser(w http.ResponseWriter, r *http.Request) {
    var user User
    if err := json.NewDecoder(r.Body).Decode(&user); err != nil {
        http.Error(w, err.Error(), 400)
        return
    }
    
    mu.Lock()
    user.ID = nextID
    nextID++
    users[user.ID] = user
    mu.Unlock()
    
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(201)
    json.NewEncoder(w).Encode(user)
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("/users", func(w http.ResponseWriter, r *http.Request) {
        switch r.Method {
        case "GET":
            getUsers(w, r)
        case "POST":
            createUser(w, r)
        }
    })
    
    fmt.Println("REST API启动在 :8080")
    http.ListenAndServe(":8080", mux)
}
```

### 挑战题: RESTful用户API

```go
package main

import (
    "encoding/json"
    "fmt"
    "net/http"
    "sync"
)

type User struct {
    ID   int    `json:"id"`
    Name string `json:"name"`
    Age  int    `json:"age"`
}

var (
    users = make(map[int]User)
    nextID = 1
    mu     sync.RWMutex
)

func JSON(w http.ResponseWriter, status int, data interface{}) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(data)
}

func listUsers(w http.ResponseWriter, r *http.Request) {
    mu.RLock()
    defer mu.RUnlock()
    
    result := []User{}
    for _, u := range users {
        result = append(result, u)
    }
    JSON(w, 200, result)
}

func getUser(w http.ResponseWriter, r *http.Request) {
    var id int
    fmt.Sscanf(r.URL.Path, "/api/users/%d", &id)
    
    mu.RLock()
    defer mu.RUnlock()
    
    if user, ok := users[id]; ok {
        JSON(w, 200, user)
    } else {
        JSON(w, 404, map[string]string{"error": "not found"})
    }
}

func createUser(w http.ResponseWriter, r *http.Request) {
    var user User
    if err := json.NewDecoder(r.Body).Decode(&user); err != nil {
        JSON(w, 400, map[string]string{"error": "invalid request"})
        return
    }
    
    mu.Lock()
    user.ID = nextID
    nextID++
    users[user.ID] = user
    mu.Unlock()
    
    JSON(w, 201, user)
}

func main() {
    http.HandleFunc("/api/users", func(w http.ResponseWriter, r *http.Request) {
        switch r.Method {
        case "GET":
            listUsers(w, r)
        case "POST":
            createUser(w, r)
        default:
            JSON(w, 405, map[string]string{"error": "method not allowed"})
        }
    })
    
    http.HandleFunc("/api/users/", getUser)
    
    fmt.Println("REST API: http://localhost:8080/api/users")
    http.ListenAndServe(":8080", nil)
}
```
